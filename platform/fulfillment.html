⚙️ Intégration PHP — Comment envoyer les commandes

Dans ton script PHP de confirmation de commande, envoie un POST à :

https://v35-fulfillment.ernestpedanou.workers.dev/orders 
// PHP — après validation paiement
$order = [
  'shop'     => 'ma-boutique.fr',
  'email'    => $customer['email'],
  'customer' => ['name'=>$customer['name'], 'city'=>$customer['city'], 'country'=>'FR'],
  'shipping' => ['address1'=>$addr, 'city'=>$city, 'zip'=>$zip, 'country'=>'France'],
  'currency' => 'EUR',
  'items'    => array_map(fn($p) => [
    'title'    => $p['title'],
    'quantity' => $p['qty'],
    'price'    => $p['price'],
    'sku'      => $p['sku'],
  ], $cart),
];
$ch = curl_init('https://v35-fulfillment.ernestpedanou.workers.dev/orders');
curl_setopt_array($ch, [
  CURLOPT_POST => true,
  CURLOPT_POSTFIELDS => json_encode($order),
  CURLOPT_HTTPHEADER => ['Content-Type: application/json'],
  CURLOPT_RETURNTRANSFER => true,
]);
curl_exec($ch);
